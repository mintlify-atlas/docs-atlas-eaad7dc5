## Overview

Start UI [web] uses a combination of state management libraries optimized for different use cases:

- **Server State**: TanStack Query for API data
- **Client State**: Zustand for global UI state
- **Form State**: React Hook Form for form data
- **UI State**: React hooks and context for component-level state

## Server State with TanStack Query

TanStack Query (React Query) handles server state, caching, and synchronization.

### Query Client Setup

```typescript src/lib/tanstack-query/query-client.ts
import { QueryClient } from '@tanstack/react-query';

export const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 10, // 10 seconds
      retry: 1,
    },
  },
});
```

### Using Queries

Fetch data with automatic caching and refetching:

```typescript
import { useQuery } from '@tanstack/react-query';
import { orpcClient } from '@/lib/orpc/client';

function BookList() {
  const { data, isLoading, error } = useQuery({
    queryKey: ['books'],
    queryFn: () => orpcClient.book.list({}),
  });

  if (isLoading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  return (
    <ul>
      {data?.items.map((book) => (
        <li key={book.id}>{book.title}</li>
      ))}
    </ul>
  );
}
```

### Using Mutations

Modify server data with optimistic updates:

```typescript
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { orpcClient } from '@/lib/orpc/client';

function CreateBookButton() {
  const queryClient = useQueryClient();
  
  const mutation = useMutation({
    mutationFn: (data: BookInput) => orpcClient.book.create(data),
    onSuccess: () => {
      // Invalidate and refetch
      queryClient.invalidateQueries({ queryKey: ['books'] });
    },
  });

  return (
    <button 
      onClick={() => mutation.mutate({ 
        title: 'New Book',
        author: 'John Doe'
      })}
      disabled={mutation.isPending}
    >
      {mutation.isPending ? 'Creating...' : 'Create Book'}
    </button>
  );
}
```

### Pagination

Handle paginated data:

```typescript
import { useQuery } from '@tanstack/react-query';
import { orpcClient } from '@/lib/orpc/client';

function PaginatedBooks() {
  const [page, setPage] = useState(1);
  
  const { data } = useQuery({
    queryKey: ['books', page],
    queryFn: () => orpcClient.book.list({ page, perPage: 10 }),
    placeholderData: (prev) => prev, // Keep previous data while fetching
  });

  return (
    <div>
      {data?.items.map((book) => <div key={book.id}>{book.title}</div>)}
      <button onClick={() => setPage((p) => p - 1)} disabled={page === 1}>
        Previous
      </button>
      <button onClick={() => setPage((p) => p + 1)}>
        Next
      </button>
    </div>
  );
}
```

## Client State with Zustand

Zustand provides lightweight global state management.

### Creating a Store

```typescript src/features/cart/cart-store.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface CartStore {
  items: CartItem[];
  addItem: (item: CartItem) => void;
  removeItem: (id: string) => void;
  clearCart: () => void;
}

export const useCartStore = create<CartStore>()(persist(
  (set) => ({
    items: [],
    addItem: (item) => set((state) => ({ 
      items: [...state.items, item] 
    })),
    removeItem: (id) => set((state) => ({ 
      items: state.items.filter((item) => item.id !== id) 
    })),
    clearCart: () => set({ items: [] }),
  }),
  {
    name: 'cart-storage',
  }
));
```

### Using the Store

```typescript
function ShoppingCart() {
  const items = useCartStore((state) => state.items);
  const removeItem = useCartStore((state) => state.removeItem);
  
  return (
    <div>
      {items.map((item) => (
        <div key={item.id}>
          {item.name}
          <button onClick={() => removeItem(item.id)}>Remove</button>
        </div>
      ))}
    </div>
  );
}
```

### Selector Optimization

Use selectors to prevent unnecessary re-renders:

```typescript
// ❌ Will re-render on ANY state change
const store = useCartStore();

// ✅ Only re-renders when items change
const items = useCartStore((state) => state.items);
```

## Form State with React Hook Form

React Hook Form manages form state with minimal re-renders. See the [Forms](/features/forms) documentation for detailed usage.

```typescript
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

const schema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
});

function LoginForm() {
  const form = useForm({
    resolver: zodResolver(schema),
    defaultValues: {
      email: '',
      password: '',
    },
  });

  const onSubmit = form.handleSubmit((data) => {
    console.log(data);
  });

  return (
    <form onSubmit={onSubmit}>
      <input {...form.register('email')} />
      <input type="password" {...form.register('password')} />
      <button type="submit">Submit</button>
    </form>
  );
}
```

## UI State Patterns

### Local Component State

Use `useState` for simple component state:

```typescript
import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);
  
  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
    </div>
  );
}
```

### Disclosure Pattern

Use `react-use-disclosure` for modal/drawer state:

```typescript
import { useDisclosure } from 'react-use-disclosure';

function DialogExample() {
  const dialog = useDisclosure();
  
  return (
    <div>
      <button onClick={dialog.open}>Open Dialog</button>
      
      <Dialog open={dialog.isOpen} onOpenChange={dialog.toggle}>
        <DialogContent>
          <h2>Dialog Title</h2>
          <button onClick={dialog.close}>Close</button>
        </DialogContent>
      </Dialog>
    </div>
  );
}
```

### Context for Component Trees

Share state across a component tree:

```typescript
import { createContext, useContext, useState } from 'react';

const ThemeContext = createContext<{
  theme: 'light' | 'dark';
  toggleTheme: () => void;
} | null>(null);

function ThemeProvider({ children }) {
  const [theme, setTheme] = useState<'light' | 'dark'>('light');
  
  const toggleTheme = () => {
    setTheme((prev) => (prev === 'light' ? 'dark' : 'light'));
  };
  
  return (
    <ThemeContext.Provider value={{ theme, toggleTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) throw new Error('useTheme must be used within ThemeProvider');
  return context;
}
```

## State Management Decision Tree

<Steps>
  <Step title="Is it server data?">
    Use **TanStack Query** for fetching, caching, and synchronizing server state
  </Step>
  <Step title="Is it form data?">
    Use **React Hook Form** for efficient form state with validation
  </Step>
  <Step title="Does it need to persist?">
    Use **Zustand with persist middleware** for data that survives page reloads
  </Step>
  <Step title="Is it shared across the app?">
    Use **Zustand** for global client state (theme, sidebar, notifications)
  </Step>
  <Step title="Is it local to a component tree?">
    Use **React Context** for shared state in a subtree
  </Step>
  <Step title="Is it simple component state?">
    Use **useState** for local component state
  </Step>
</Steps>
