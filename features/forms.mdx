Start UI [web] provides a comprehensive form system built with [React Hook Form](https://react-hook-form.com/) and [Zod](https://zod.dev/) for type-safe validation.

## Overview

The form system includes:

- **React Hook Form**: Performant forms with minimal re-renders
- **Zod Validation**: Type-safe schema validation
- **Custom Components**: Pre-built field components with consistent styling
- **i18n Support**: Internationalized error messages
- **Accessibility**: ARIA labels and keyboard navigation

## Quick Start

Here's a complete form example:

```tsx
import { zodResolver } from '@hookform/resolvers/zod';
import { useForm } from 'react-hook-form';
import { z } from 'zod';
import {
  Form,
  FormField,
  FormFieldController,
  FormFieldLabel,
} from '@/components/form';
import { Button } from '@/components/ui/button';

// Define schema
const schema = z.object({
  name: z.string().min(1, 'Name is required'),
  email: z.string().email('Invalid email'),
});

type FormData = z.infer<typeof schema>;

export function MyForm() {
  const form = useForm<FormData>({
    resolver: zodResolver(schema),
    defaultValues: {
      name: '',
      email: '',
    },
  });

  const onSubmit = (data: FormData) => {
    console.log(data);
  };

  return (
    <Form {...form} onSubmit={onSubmit}>
      <FormField>
        <FormFieldLabel>Name</FormFieldLabel>
        <FormFieldController
          control={form.control}
          type="text"
          name="name"
        />
      </FormField>

      <FormField>
        <FormFieldLabel>Email</FormFieldLabel>
        <FormFieldController
          control={form.control}
          type="email"
          name="email"
        />
      </FormField>

      <Button type="submit" loading={form.formState.isSubmitting}>
        Submit
      </Button>
    </Form>
  );
}
```

## Zod Schemas

Define validation schemas with Zod:

```ts src/features/auth/schema.ts
import { t } from 'i18next';
import { z } from 'zod';
import { zu } from '@/lib/zod/zod-utils';

export const zFormFieldsLogin = () =>
  z.object({
    email: zu.fieldText.required().pipe(
      z.email({
        error: (issue) =>
          issue.input
            ? t('auth:common.email.invalid')
            : t('auth:common.email.required'),
      })
    ),
  });

export type FormFieldsLogin = z.infer<ReturnType<typeof zFormFieldsLogin>>;
```

### Common Validation Patterns

```ts
// Required text field
const name = zu.fieldText.required();

// Email with custom error
const email = z
  .string()
  .min(1, 'Email is required')
  .email('Invalid email');

// Number with range
const age = z
  .number()
  .min(18, 'Must be 18 or older')
  .max(120, 'Invalid age');

// String length validation
const otp = z
  .string()
  .length(6, 'Code must be 6 characters');

// Optional field
const publisher = z.string().optional();

// Enum/Select
const role = z.enum(['user', 'admin']);

// Array of values
const tags = z.array(z.string()).min(1, 'At least one tag required');

// Date validation
const birthdate = z.date({
  required_error: 'Birthdate is required',
});
```

## Real-World Examples

### Login Form

```tsx src/features/auth/page-login.tsx
import { zodResolver } from '@hookform/resolvers/zod';
import { useForm } from 'react-hook-form';
import { useTranslation } from 'react-i18next';

const form = useForm({
  mode: 'onSubmit',
  resolver: zodResolver(zFormFieldsLogin()),
  defaultValues: {
    email: '',
  },
});

const submitHandler: SubmitHandler<FormFieldsLogin> = async ({ email }) => {
  const result = await authClient.emailOtp.sendVerificationOtp({
    email,
    type: 'sign-in',
  });

  if (result.error) {
    toast.error(t(`auth:errorCode.${result.error.code}`));
    return;
  }

  router.navigate({
    to: '/login/verify',
    search: { email },
  });
};

return (
  <Form {...form} onSubmit={submitHandler}>
    <FormField>
      <FormFieldLabel className="sr-only">
        {t('auth:common.email.label')}
      </FormFieldLabel>
      <FormFieldController
        type="email"
        control={form.control}
        name="email"
        size="lg"
        placeholder={t('auth:common.email.label')}
      />
    </FormField>
    <Button
      loading={form.formState.isSubmitting}
      type="submit"
      size="lg"
    >
      {t('auth:pageLogin.loginWithEmail')}
    </Button>
  </Form>
);
```

### Form with Drawer

```tsx src/features/account/change-name-drawer.tsx
import { zodResolver } from '@hookform/resolvers/zod';
import { useMutation } from '@tanstack/react-query';
import { useState } from 'react';
import { useForm } from 'react-hook-form';
import { toast } from 'sonner';

export const ChangeNameDrawer = ({ children }) => {
  const [open, setOpen] = useState(false);
  const session = authClient.useSession();
  
  const form = useForm<FormFieldsAccountUpdateName>({
    resolver: zodResolver(zFormFieldsAccountUpdateName()),
    values: {
      name: session.data?.user.name ?? '',
    },
  });

  const updateUser = useMutation(
    orpc.account.updateInfo.mutationOptions({
      onSuccess: async () => {
        await session.refetch();
        toast.success(t('account:changeNameDrawer.successMessage'));
        form.reset();
        setOpen(false);
      },
      onError: () => toast.error(t('account:changeNameDrawer.errorMessage')),
    })
  );

  return (
    <ResponsiveDrawer
      open={open}
      onOpenChange={(isOpen) => {
        setOpen(isOpen);
        form.reset();
      }}
    >
      <ResponsiveDrawerTrigger render={children} />
      <ResponsiveDrawerContent>
        <Form
          {...form}
          onSubmit={({ name }) => updateUser.mutate({ name })}
        >
          <ResponsiveDrawerHeader>
            <ResponsiveDrawerTitle>
              {t('account:changeNameDrawer.title')}
            </ResponsiveDrawerTitle>
          </ResponsiveDrawerHeader>
          
          <ResponsiveDrawerBody>
            <FormField>
              <FormFieldLabel className="sr-only">
                {t('account:changeNameDrawer.label')}
              </FormFieldLabel>
              <FormFieldController
                control={form.control}
                type="text"
                name="name"
                size="lg"
                autoFocus
              />
            </FormField>
          </ResponsiveDrawerBody>
          
          <ResponsiveDrawerFooter>
            <Button
              type="submit"
              loading={updateUser.isPending}
            >
              {t('account:changeNameDrawer.submitButton')}
            </Button>
          </ResponsiveDrawerFooter>
        </Form>
      </ResponsiveDrawerContent>
    </ResponsiveDrawer>
  );
};
```

## Form State

### Form Validation Modes

```tsx
const form = useForm({
  mode: 'onSubmit', // Validate on submit (default)
  // mode: 'onChange', // Validate on every change
  // mode: 'onBlur', // Validate on blur
  // mode: 'onTouched', // Validate after first blur
});
```

### Check Form State

```tsx
const { formState } = form;

// Is form valid?
const isValid = formState.isValid;

// Is form submitting?
const isSubmitting = formState.isSubmitting;

// Was form submitted?
const isSubmitted = formState.isSubmitted;

// Are there errors?
const hasErrors = Object.keys(formState.errors).length > 0;

// Is field dirty (changed)?
const isDirty = formState.dirtyFields.name;
```

### Watch Field Values

```tsx
import { useWatch } from 'react-hook-form';

// Watch a single field
const email = useWatch({
  control: form.control,
  name: 'email',
});

// Watch multiple fields
const [name, email] = useWatch({
  control: form.control,
  name: ['name', 'email'],
});
```

### Display Field Errors

    name="email"
  />

### Manual Error Setting

```tsx
const onSubmit = async (data: FormData) => {
  const result = await submitData(data);
  
  if (result.error) {
    // Set field error
    form.setError('email', {
      type: 'manual',
      message: 'This email is already taken',
    });
    
    // Or set form-level error
    form.setError('root', {
      type: 'manual',
      message: 'Something went wrong',
    });
  }
};
```

## Advanced Patterns

### Dynamic Fields

```tsx
import { useFieldArray } from 'react-hook-form';

const { fields, append, remove } = useFieldArray({
  control: form.control,
  name: 'authors',
});

return (
  <>
    {fields.map((field, index) => (
      <FormField key={field.id}>
        <FormFieldController
          control={form.control}
          type="text"
          name={`authors.${index}.name`}
        />
        <Button onClick={() => remove(index)}>Remove</Button>
      </FormField>
    ))}
    <Button onClick={() => append({ name: '' })}>Add Author</Button>
  </>
);
```

## Troubleshooting

<AccordionGroup>
  <Accordion title="Form not validating">
    Ensure you've passed the `resolver` to `useForm`:
    
    ```tsx
    const form = useForm({
      resolver: zodResolver(schema),
    });
    ```
  </Accordion>

  <Accordion title="Errors not displaying">
    Make sure you're using `<FormFieldError />` or accessing `formState.errors`:
    
    ```tsx
    <FormFieldError />
    {/* or */}
    {form.formState.errors.email?.message}
    ```
  </Accordion>

  <Accordion title="Values not updating">
    Use `watch` or `useWatch` to track field changes:
    
    ```tsx
    const value = useWatch({ control: form.control, name: 'field' });
    ```
  </Accordion>
</AccordionGroup>
